164bool encryptFile(const QString &filePath, const QString &password, QObject *parent) {
165 QFile file(filePath);
166 if (!file.open(QIODevice::ReadOnly)) {
167 qDebug() << "Ошибка открытия файла для шифрования";
168 return false;
169 }
170
171 QByteArray fileData = file.readAll();
172 file.close();
173
174 // Генерация соли
175 unsigned char salt[SALT_SIZE];
176 RAND_bytes(salt, sizeof(salt)); // Генерация случайной соли
177
178 // Логируем соль для отладки
179 qDebug() << "Соль:" << QByteArray(reinterpret_cast<char*>(salt), SALT_SIZE).toHex();
180
181 // Генерация ключа из пароля и соли
182 unsigned char key[KEY_SIZE];
183 int iterations = 10000;
184 if (!PKCS5_PBKDF2_HMAC(password.toUtf8().data(), password.size(), salt, sizeof(salt), iterations, EVP_sha256(), KEY_SIZE, key)) {
185 qDebug() << "Ошибка генерации ключа";
186 return false;
187 }
188
189 // Логируем ключ
190 qDebug() << "Ключ:" << QByteArray(reinterpret_cast<char*>(key), KEY_SIZE).toHex();
191
192 // Инициализация шифрования
193 EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
194 EVP_EncryptInit_ex(ctx, EVP_aes_256_ecb(), nullptr, key, nullptr); // AES-256-ECB, без IV
195
196 QByteArray encryptedData;
197 int blockSize = fileData.size() + AES_BLOCK_SIZE; // Добавляем запас на финальный блок
198
199 encryptedData.resize(blockSize);
200
201 int out_len = 0, final_len = 0;
202
203 // Шифрование данных
204 if (!EVP_EncryptUpdate(ctx, reinterpret_cast<unsigned char*>(encryptedData.data()), &out_len,
205 reinterpret_cast<const unsigned char*>(fileData.data()), fileData.size())) {
206 qDebug() << "Ошибка при шифровании данных";
207 EVP_CIPHER_CTX_free(ctx);
208 return false;
209 }
210
211 // Финальный блок
212 if (!EVP_EncryptFinal_ex(ctx, reinterpret_cast<unsigned char*>(encryptedData.data()) + out_len, &final_len)) {
213 qDebug() << "Ошибка при завершении шифрования";
214 EVP_CIPHER_CTX_free(ctx);
215 return false;
216 }
217
218 encryptedData.resize(out_len + final_len); // Устанавливаем финальный размер зашифрованных данных
219
220 EVP_CIPHER_CTX_free(ctx);
221
222 // Открытие файла для записи зашифрованных данных
223 if (!file.open(QIODevice::WriteOnly)) {
224 qDebug() << "Ошибка открытия файла для записи зашифрованных данных";
225 return false;
226 }
227
228 // Запись соли перед зашифрованными данными
229 file.write(reinterpret_cast<char*>(salt), SALT_SIZE);
230 file.write(encryptedData);
231 file.close();
232
233 return true;
234}

245bool decryptFile(const QString &filePath, const QString &password, QObject *parent) {
246 QFile file(filePath);
247 if (!file.open(QIODevice::ReadOnly)) {
248 qDebug() << "Ошибка открытия файла для расшифровки";
249 return false;
250 }
251
252 // Чтение соли из файла
253 unsigned char salt[SALT_SIZE];
254 if (file.read(reinterpret_cast<char*>(salt), SALT_SIZE) != SALT_SIZE) {
255 qDebug() << "Ошибка чтения соли";
256 return false;
257 }
258
259 // Логируем соль для отладки
260 qDebug() << "Соль (при расшифровке):" << QByteArray(reinterpret_cast<char*>(salt), SALT_SIZE).toHex();
261
262 // Чтение зашифрованных данных
263 QByteArray encryptedData = file.readAll();
264 file.close();
265
266 // Генерация ключа из пароля и соли
267 unsigned char key[KEY_SIZE];
268 int iterations = 10000;
269 if (!PKCS5_PBKDF2_HMAC(password.toUtf8().data(), password.size(), salt, sizeof(salt), iterations, EVP_sha256(), KEY_SIZE, key)) {
270 qDebug() << "Ошибка генерации ключа";
271 return false;
272 }
273
274 // Логируем ключ для отладки
275 qDebug() << "Ключ (при расшифровке):" << QByteArray(reinterpret_cast<char*>(key), KEY_SIZE).toHex();
276
277 // Инициализация расшифрования
278 EVP_CIPHER_CTX *ctx = EVP_CIPHER_CTX_new();
279 EVP_DecryptInit_ex(ctx, EVP_aes_256_ecb(), nullptr, key, nullptr); // AES-256-ECB, без IV
280
281 QByteArray decryptedData;
282 int blockSize = encryptedData.size(); // Размер буфера для расшифрованных данных
283 decryptedData.resize(blockSize);
284
285 int out_len = 0, final_len = 0;
286
287 // Расшифрование данных
288 if (!EVP_DecryptUpdate(ctx, reinterpret_cast<unsigned char*>(decryptedData.data()), &out_len,
289 reinterpret_cast<const unsigned char*>(encryptedData.data()), encryptedData.size())) {
290 qDebug() << "Ошибка при расшифровке данных";
291 EVP_CIPHER_CTX_free(ctx);
292 return false;
293 }
294
295 // Финальный блок
296 if (EVP_DecryptFinal_ex(ctx, reinterpret_cast<unsigned char*>(decryptedData.data()) + out_len, &final_len) != 1) {
297 qDebug() << "Ошибка при завершении расшифровки. Неверный пароль или поврежденные данные";
298 EVP_CIPHER_CTX_free(ctx);
299 return false;
300 }
301
302 decryptedData.resize(out_len + final_len); // Обрезаем данные до корректного размера
303
304 EVP_CIPHER_CTX_free(ctx);
305
306 // Открытие файла для записи расшифрованных данных
307 if (!file.open(QIODevice::WriteOnly)) {
308 qDebug() << "Ошибка открытия файла для записи расшифрованных данных";
309 return false;
310 }
311
312 file.write(decryptedData);
313 file.close();
314
315 return true;
316}
